	yum install httpd  // cài đặt gói tin http apache
	
	# các bước để bật dịch vụ http trên hệ thống 
	
	1. Bước 1 cài đặt dịch vụ httpd 
	
	systemctl status httpd // kieêm tra trang thái ( dead =  chưa bật ) 
	
	systemctl enable httpd //  kích hoạt cài dv
	systemctl start httpd // bật dịch vụ 
	
	2. Bước 2 cài đặt netstart 
	
	yum install net-tools 
	
	netstat -tlpn  ( t : tcp  | l : listen | p : port |  n : number )   // check xem hiện dịch vụ đang lắng nghe cổng nào 
	

	
	 3. cấu hình dịch vụ apache( giống lúc cài web hôm nọ ) 
	 # /etc/httpd/conf/httpd.conf    -- file chứa các thông tin cấu hình ( vd đổi port thì Listen 80 -> 8080
	 
	 -> tìm đến cuối  -/Document Root "/var/www/html/" ///  địa chỉ của nó là thư mục chứa nội dung trang web 
	 
	 cd /var/www/html/  // truy cập vào thư mục chưa web
