Lesson 3 -- -- -- Hệ thống lệnh Linux -//


	
	- /  :    // cú pháp tìm kiếm trong vi , trong trang tìm kiếm ( /user )
	
	useradd nameUser 		: // tạo 1 user mới 
	
	- mkdir 				// tạo thư mục 
			mkdir -p oldFolder/newFolder/oldFolder   // chèn thêm 1 thư mục vào ( -p : parent)
			mkdir folder{1,2,3..}   // tạo ra 3 thư mục folder1 ,folder2, folder 3
			mkdir folder1 folder2 folder3 'folder 4'   // tạo ra nhiều thư mục   
			
	- rmdir		// xóa thư mục trống
			rmdir folder{1,2,3..}   // xóa ra 3 thư mục folder1 ,folder2, folder 3
			
	- rm -r  name	// xóa thư mục & file không nhân nhượng 
		rm -r folder{1,2,3..}
		rm -rf  folder{1,2..}  // xóa mà không cần hỏi 
		
	- touch /root/folder/{file1,'file 2' , file3}  ... // tạo nhiều file tại vị trí khác 
	
	
	- cp   // copy
			cp /root/folder1/fileName /root/folder2 	//copy file đến thư mục khác 
			cp -r /root/folder /etc/folder		// copy thư mục phải thêm option 
			cp -rv /root/folder /etc/folder		// copy thư mục và hiển thị quá trình copy
			
	- dd   // copy với số lượng có hạn 
		vd :   dd if=/root/1file.txt of=/root/2file.txt bs=512 count=1024  // 
					if -> input file 
					of -> output file 
					bs -> bit size 
					count -> block count ( số lượng ký tự ) 
	   
	- mv 	// move 
	
# Đọc file 
	
	- cat   // đọc file    cat /etc/passwd
	
	- more	// đọc file    more /etc/passwd
	
	*- less	// đọc file    less /etc/passwd
			G : xuống cuối  g về đầu 
			/findText : tìm kiếm  ->  N :  đi tiếp kết quả ;  n : lùi về kq trước
			q  :  exit
			
	- head   // đọc 10 dòng đầu 
			head -n5 /etc/passwd   : đọc 5 dòng đầu 
	*- tail  // đọc phần đuôi 
			tail -n1 /etc/passwd
			tail -f  /etc/passwd   // đọc file log vừa được thay đổi  
		
#  tìm kiếm file 
	- find / -name root   // tìm kiếm file theo tên 
						*.txt 	// tìm kiếm file có đuôi là txt
		     -size 1000k  // file có size == 1000k 
					   -1000k  // file có dung lượng nhỏ hơn 
					   +1000k // file có size lớn hơn 1000k
					   
			  -ctime -2   // file có change time < 2 ngày 
				 
		  //e.g kết hợp 
		   find /etc -name root -size +1000k -ctime -100
		   
		   ===> 
		   find / "*.txt" |  xargs rm -rf  // xóa tất cả cácfile nó tìm được , xargs chuyển các kết quả tìm được thành đối số của lệnh rm -> tiện ha
