# Ohter comnand-- /  out put ->
	*expand  // chuyển các dấu tab thành các khoảng trắng         -> expand -t 4 file.txt  ( chuyển dấy tab thành 4 dấu cách ) 
	*unexpand  // chuyển các khoảng trắng thành các dấu tab 		-> unexpand -a file txt 
	*paste    // kết hợp nhiều tập tin văn bản thành một văn bản duy nhất   paste option inputfile1 inputfile2   -> paste -d":" file1 file2  ( ngăn cách các phần tử bới dấu : 
	*join   // tập hợp dữ liệu của 2 dữ liệu trên một trường chung 			join option file1 file1   -> join -j 2 file1 file1
	*split	// chia nhỏ một tập tin thành một tập tin nhỏ hơn           split option inputfile        -> split -l 2 file.txt -> cắt thành các file xaa xab xac  ( 2 =  cắt 2 dòng 1 ) 
	*nl			// number line đánh số dòng 							nl option inputfile 
	*pr			// in ra nội dung văn bản theo dạng cột 				pr option inputfile 			-> pr -n file.txt
	*sort		// sắp xếp 												sort option inputfile    ( -M : moon (xuôi) , -n : XUÔI , -r : ngược ) sort -r file.txt
	*uniq		//  loại bỏ các dòng trùng lặp 									-> uniq -c file.txt
	*fmt      //định dạng lại nội dung để các dòng có width giống nhau 			-> fmt -w 30 text.txt ( tối đa in ra 30 ký tự ko hơn trên 1 dòng ) 
	*tr      // thay thế hoặc xóa các ký tự trong một tập tin (replace)			-> cat test.txt | tr -d t  ( thay thế ký tự t = "") || cat test.txt | tr "[:lower:]" "[:upper:]" ( thay thế ký tự thường = hoa )  
	*od		// xuất ra theo dạng tiêu chuẩn bát phân hoặc ÁCII  với file có nội dung nhiều hình thức    -> od -o text.txt ( chuyển thành dạng bát phân )   || -o :8 , -i:10 , -x:16 , -c:ASCII ) 
	*sed	// lệnh dùng để sửa 				--> ( a : add ,  i : insert , d : delete ,  s/条件/ : sửa theo điều kiện   -> sed '1d' text.txt  : Xóa dòng số 1	
	*wc    // write count
