Lesson 21 -- package -//

		
	- sử dụng để import nhiều module trong thư mục khác 
	- packgae là một folder chưa các module , packagecon(sub package) và bên cạnh đó là file __init__.py (tức là bắt buộc phải có __init__.py để python chạy và nhận ra được đây là package chứ k phải một thư mục thông thường)
	- file __init__.py được chạy khi chúng ta import , chúng ta có thể đặt một vài dòng debug hoặc thông báo hoặc xử lý gì đó tại đây để dễ kiểm soát luồng import 
	- file __init__.py có thể dùng để khởi tạo các giá trị cho package  hoặc định nghĩa các đối tượng cần import
	
	
	
	# e.g1 : ví dụ cấu trúc trong một thư mục package 
	my_package/
	│
	├── __init__.py           # Tệp xác định thư mục là package
	├── module_a.py           # Mô-đun trong package
	└── sub_package/
		├── __init__.py       # Tệp init cho subpackage
		└── module_b.py       # Mô-đun trong subpackage
	
	
	
	# e.g 2 : Sử dụng import toàn bộ module hoặc pakage vào namespace hiện tại 
	import my_package.module_a
	my_package.module_a.some_function()			# gọi hàm bên trong module_a
	
	
	# e.g 3 : Import trực tiếp hàm , biến ,  lớp 
	
	from my_package.module_a import some_function 
	some_function()   				# gọi hàm trực tiếp 
	
	# e.g 4 : import tất cả nội dung module 
	from my_package.module_a import * 
	
	
	# e.g 5 : sử dụng alias (Biệt danh ) khi import 
	import my_package.module_a as mod_a
	mod_a.some_function()
	
	# e.g 6 : import package con(sub package )
	from my_package.sub_package import module_b
	module_b.some_function_in_b()
	
	
	
	# e.g 7 : import có điều kiện 
	try : 
		import my_package.module_a
	except ImportError:
		print("Module not found")
		
		
		
	# e.g 8: tìm hiểu đường dẫn tìm kiếm(Module Search Path ) , python tim các module theo sys.path 
	import sys
	print(sys.path)					# in ra các đường dẫn ['D:\\Dev\\Python\\Learn\\Project', 'C:\\Users\\USER\\AppData\\Local\\Programs\\Python\\Python313\\python313.zip', ....]
	
	
	
	# e.g 9 : cách import các module trong cấu trúc phức tạp 
						my_complex_package/
							│
							├── __init__.py
							├── core/
							│   ├── __init__.py
							│   └── utils.py
							└── data/
								├── __init__.py
								└── loader.py

		from my_complex_package.core import utils
		from my_complex_package.data import loader
		
	
	# e.g 10  : Cách tự tạo và cài đặt package với pip
		tạo file setup.py cho package và dùng [pip install .] để cài đặt
		
	from setuptools import setup, find_package
	
	setup(
		name="my_package",
		version="0.1"
		packages=find_packages(),
	)
