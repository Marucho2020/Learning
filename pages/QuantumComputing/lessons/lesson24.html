# 3. üßÆ M√¥ h√¨nh h√≥a b·∫±ng to√°n h·ªçc (Kraus Operators)

C√°c noise ƒë∆∞·ª£c m√¥ t·∫£ b·∫±ng Kraus operators {E_i},√°p d·ª•ng l√™n tr·∫°ng th√°i:

import numpy as np

I = np.eye(2)
X = np.array([[0, 1], [1, 0]])

p = 0.2  # X√°c su·∫•t b·ªã l·∫≠t bit

E0 = np.sqrt(1 - p) * I
E1 = np.sqrt(p) * X

# tr·∫°ng th√°i ban ƒë·∫ßu |0‚ü©
ket0 = np.array([[1], [0]])
rho = ket0 @ ket0.T.conj()

# √Åp d·ª•ng noise: rho' = E0 œÅ E0‚Ä† + E1 œÅ E1‚Ä†
rho_prime = E0 @ rho @ E0.T.conj() + E1 @ rho @ E1.T.conj()
print("After Bit-flip noise:\n", rho_prime)

# 5. üß¨ M√¥ ph·ªèng Noise trong Qiskit
from qiskit.providers.aer.noise import NoiseModel, depolarizing_error
from qiskit import QuantumCircuit, Aer, transpile, execute

# M·∫°ch ƒë∆°n gi·∫£n
qc = QuantumCircuit(1, 1)
qc.h(0)
qc.measure(0, 0)

# T·∫°o noise model: 5% depolarizing noise sau H gate
noise_model = NoiseModel()
error = depolarizing_error(0.05, 1)
noise_model.add_all_qubit_quantum_error(error, ['h'])

# Run simulation
backend = Aer.get_backend('qasm_simulator')
job = execute(qc, backend, noise_model=noise_model, shots=1000)
result = job.result()
print(result.get_counts())
