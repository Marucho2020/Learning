# 3. üìò Code v√≠ d·ª• Kronecker b·∫±ng NumPy 
import numpy as np

# Hai vector qubit |0‚ü© v√† |1‚ü©
zero = np.array([1, 0])
one = np.array([0, 1])

# Tr·∫°ng th√°i 2 qubit: |0‚ü©‚äó|1‚ü©

print("Tensor |0‚ü©‚äó|1‚ü© =", tensor_state)
# K·∫øt qu·∫£: [0 1 0 0] ‚Üí ch·ªâ c√≥ ph·∫ßn t·ª≠ th·ª© 2 l√† 1 (v·ªã tr√≠ |01‚ü©)

T·∫°o 3 qubit:

print("Tensor |0‚ü©‚äó|1‚ü©‚äó|1‚ü© =", state)


# 4. ‚ö°·ª®ng d·ª•ng: Tr·∫°ng th√°i qubit nhi·ªÅu qubit trong l∆∞·ª£ng t·ª≠

# 5. ‚ú® Tensor product v·ªõi quantum gates
T∆∞∆°ng t·ª± vector, ma tr·∫≠n gate c≈©ng d√πng tensor.

V√≠ d·ª•: ma tr·∫≠n I v√† X:

              [0, 1]])

              [1, 0]])

# 6. üß™ Th·ª±c h√†nh mini b√†i h√¥m nay
# T·∫°o tr·∫°ng th√°i 2 qubit |0‚ü©‚äó|1‚ü© v√† 3 qubit |0‚ü©‚äó|1‚ü©‚äó|0‚ü©

# |010‚ü©
