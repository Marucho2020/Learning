Khai báo và các hằng số trong TimeUnit

	
	public enum TimeUnit {
    NANOSECONDS,    // Nano giây (10^-9 giây)
    MICROSECONDS,   // Micro giây (10^-6 giây)
    MILLISECONDS,   // Mili giây (10^-3 giây)
    SECONDS,        // Giây
    MINUTES,        // Phút
    HOURS,          // Giờ
    DAYS            // Ngày
}
