LocalDate - Làm việc với ngày

	
	LocalDate đại diện cho ngày tháng, nhưng không chứa thông tin về giờ, phút, giây.
	
		import java.time.LocalDate;
		public class LocalDateExample {
			public static void main(String[] args) {
				LocalDate today = LocalDate.now(); // Lấy ngày hiện tại
				LocalDate specificDate = LocalDate.of(2025, 2, 5); // Tạo ngày cụ thể
		
				System.out.println("Hôm nay: " + today);
				System.out.println("Ngày cụ thể: " + specificDate);
			}
		}
		
		
		import java.time.LocalDate;
		public class LocalDateMethods {
			public static void main(String[] args) {
				LocalDate today = LocalDate.now();
				
				LocalDate tomorrow = today.plusDays(1); // Cộng 1 ngày
				LocalDate lastMonth = today.minusMonths(1); // Lùi 1 tháng
				LocalDate nextYear = today.plusYears(1); // Cộng 1 năm
				
				System.out.println("Ngày mai: " + tomorrow);
				System.out.println("Tháng trước: " + lastMonth);
				System.out.println("Năm sau: " + nextYear);
		
				System.out.println("Ngày trong tuần: " + today.getDayOfWeek()); // Thứ mấy
				System.out.println("Ngày trong tháng: " + today.getDayOfMonth()); // Ngày số mấy
				System.out.println("Năm nhuận? " + today.isLeapYear()); // Có phải năm nhuận không
			}
		}


3. LocalTime -  Làm việc với thời gian 
	
	LocalTime chỉ lưu giờ, phút, giây, không chứa ngày tháng.	


4. LocalDateTime – Kết hợp ngày & giờ 
	
	LocalDateTime là sự kết hợp của LocalDate và LocalTime.
	
	
5. 	ZonedDateTime – Làm việc với múi giờ
		ZonedDateTime lưu trữ cả ngày, giờ & múi giờ.
	

6. Instant – Thời gian tính từ Epoch (UNIX Time)
	Instant đại diện cho thời gian tính theo UTC, thường dùng trong hệ thống lưu trữ timestamp.
	
7. Duration & Period – Khoảng cách giữa hai thời điểm
	Duration (đo thời gian bằng giây, mili giây)

		
	Period (đo khoảng cách bằng ngày, tháng, năm)
			import java.time.Period;
			import java.time.LocalDate;
			
			public class PeriodExample {
				public static void main(String[] args) {
					LocalDate startDate = LocalDate.of(2023, 1, 1);
					LocalDate endDate = LocalDate.of(2025, 2, 5);
			
					Period period = Period.between(startDate, endDate);
					System.out.println("Khoảng cách: " + period.getYears() + " năm, " + period.getMonths() + " tháng, " + period.getDays() + " ngày");
				}
			}
