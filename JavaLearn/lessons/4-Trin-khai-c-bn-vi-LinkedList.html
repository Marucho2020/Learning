4 Triển khai cơ bản với LinkedList

	
	Queue	queue = new LinkedList<>();
	queue.offer("Alice");
	queue.offer("Bob");
	queue.offer("Charlie");
	
	//Hiển thị hàng đợi 
	System.out.println(queue);
	// Truy xuất phần tử đầu tiên 
	System.out.println(queue.peek());
	// Loại bỏ phần tử đầu tiên 
	System.out.println(queue.poll());
