2 Cách sử dụng TreeSet

	TreeSet treeSet = new TreeSet<>();
	
	treeSet.add(15);
	treeSet.add(10);
	treeSet.add(5);
	
	// duyệt 
	for(int num : treeSet){
		System.out.println(num);			// 5 10 15
	}
	
	// lấy phần tử đầu và cuối -> phần tử lớn nhất và nhỏ nhất vì nó đã được sắp xếp  
	treeSet.first();
	treeSet.last();
	
	// Xóa phần tử theo giá trị truyền vào 
	treeSet.remove(10);
