Biến static

	Trong class , biến static là biến dùng chung cho tất cả các instance, một instance mới được tạo ra và thay đổi giá trị static này thì toàn bộ các biến instance khác cũng sẽ bị thay đổi giá trị của biến static này theo  
		# Ví dụ 
			class App {
					private static int v1; // Biến static v1 dùng chung giữa tất cả các đối tượng
					private int v2; // Biến instance v2 riêng biệt cho từng đối tượng
					
					// Constructor không tham số, gọi constructor có tham số với giá trị mặc định 20 và 40
					public App() {
						this(20, 40);
					}
					
					// Constructor nhận tham số, gán giá trị cho v1 và v2
					public App(int v1, int v2) {
						this.v1 = v1; // Gán giá trị cho biến static v1
						this.v2 = v2; // Gán giá trị cho biến instance v2
					}
					
					// Phương thức in giá trị của v1 và v2
					public void func() {
						System.out.println(v1 + " : " + v2 + " : ");
					}
					
					class Test {
						public static void main(String[] args) {
						App a1 = new App(10, 30); // Tạo đối tượng App với v1 = 10, v2 = 30
						App a2 = new App(); // Tạo đối tượng App với giá trị mặc định v1 = 20, v2 = 40
					
						// Gọi phương thức func() của cả hai đối tượng
						a1.func(); // In giá trị v1 và v2 của đối tượng a1
						a2.func(); // In giá trị v1 và v2 của đối tượng a2
						}
					}
					}
					// Kết quả là 20 30 20 40 bởi vì:
					// - v1 là biến static, dùng chung giữa tất cả các đối tượng App. Khi khởi tạo a1, v1 được gán giá trị 10.
					//   Sau đó, khi khởi tạo a2, v1 bị ghi đè với giá trị 20 (từ constructor của a2).
					// - v2 là biến instance, lưu trữ giá trị riêng biệt cho từng đối tượng. Do đó, giá trị v2 của a1 là 30, còn của a2 là 40.
